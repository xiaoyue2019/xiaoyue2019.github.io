def halve(_list,n):
    n=int(len(_list)/n)   # list数除以线程数=份数
    for i in range(0,len(_list),n):
        yield(_list[i:i+n])

num=0
for i in halve([i for i in range(13000,14000)],50):
    exec("_thread{}=_thread({})".format(num,i))
    exec('_thread{}.start()'.format(num))
